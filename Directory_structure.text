src/main/java/com/board/api
│
├── global                          # 1. 전역 설정 및 공통 컴포넌트
│   ├── config                      # 보안, Swagger, JPA 등 각종 설정 클래스
│   │   ├── SecurityConfig.java
│   │   ├── SwaggerConfig.java
│   │   └── JpaAuditConfig.java
│   ├── common                      # 공통 응답, 유틸리티, 공통 상속용 엔티티
│   │   ├── BaseTimeEntity.java     # 모든 테이블의 생성/수정 시간 공통 처리
│   │   └── ApiResponse.java        # 통일된 JSON 응답 규격
│   ├── error                       # 예외 처리 및 에러 코드
│   │   ├── GlobalExceptionHandler.java
│   │   └── ErrorCode.java
│   └── security                    # 인증/인가 핵심 로직 (JWT 관련)
│       ├── JwtProvider.java
│       └── JwtAuthenticationFilter.java
│
└── domain                          # 2. 비즈니스 핵심 도메인 (기능별)
    │
    ├── member                      # [회원 도메인]
    │   ├── controller              # MemberController.java
    │   ├── service                 # MemberService.java
    │   ├── repository              # MemberRepository.java
    │   ├── entity                  # Member.java
    │   └── dto                     # SignupRequest, LoginRequest 등
    │
    ├── post                        # [게시글 도메인]
    │   ├── controller              # PostController.java
    │   ├── service                 # PostService.java
    │   ├── repository              # PostRepository.java
    │   ├── entity                  # Post.java
    │   └── dto                     # PostCreateRequest, PostResponse 등
    │
    └── comment                     # [댓글 도메인]
        ├── controller              # CommentController.java
        ├── service                 # CommentService.java
        ├── repository              # CommentRepository.java
        ├── entity                  # Comment.java
        └── dto                     # CommentRequest 등
